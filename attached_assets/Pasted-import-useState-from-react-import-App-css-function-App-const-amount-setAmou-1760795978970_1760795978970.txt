import { useState } from "react";
import "./App.css";

function App() {
  const [amount, setAmount] = useState("");
  const [rate, setRate] = useState("");
  const [years, setYears] = useState("");
  const [results, setResults] = useState(null);

  const calculateLoan = () => {
    const P = parseFloat(amount);
    const annualRate = parseFloat(rate);
    const Y = parseFloat(years);

    if (!P || !annualRate || !Y || P <= 0 || annualRate <= 0 || Y <= 0) {
      alert("Please enter valid positive numbers for all fields.");
      return;
    }

    const r = annualRate / 100 / 12; // monthly interest rate
    const n = Y * 12; // total number of payments

    const monthlyPayment = (P * r * Math.pow(1 + r, n)) / (Math.pow(1 + r, n) - 1);
    const totalPayment = monthlyPayment * n;
    const totalInterest = totalPayment - P;

    setResults({
      monthlyPayment: monthlyPayment.toFixed(2),
      totalPayment: totalPayment.toFixed(2),
      totalInterest: totalInterest.toFixed(2),
    });
  };

  const resetForm = () => {
    setAmount("");
    setRate("");
    setYears("");
    setResults(null);
  };

  return (
    <div className="container">
      <h1>Loan & Mortgage Calculator</h1>

      <div className="card">
        <div className="input-group">
          <label>Loan Amount ($)</label>
          <input
            type="number"
            value={amount}
            onChange={(e) => setAmount(e.target.value)}
            placeholder="e.g. 100000"
          />
        </div>

        <div className="input-group">
          <label>Annual Interest Rate (%)</label>
          <input
            type="number"
            value={rate}
            onChange={(e) => setRate(e.target.value)}
            placeholder="e.g. 5"
          />
        </div>

        <div className="input-group">
          <label>Loan Term (Years)</label>
          <input
            type="number"
            value={years}
            onChange={(e) => setYears(e.target.value)}
            placeholder="e.g. 30"
          />
        </div>

        <div className="buttons">
          <button onClick={calculateLoan}>Calculate</button>
          <button className="reset" onClick={resetForm}>
            Reset
          </button>
        </div>

        {results && (
          <div className="results">
            <h3>Results</h3>
            <p>
              <strong>Monthly Payment:</strong> ${results.monthlyPayment}
            </p>
            <p>
              <strong>Total Payment:</strong> ${results.totalPayment}
            </p>
            <p>
              <strong>Total Interest:</strong> ${results.totalInterest}
            </p>
          </div>
        )}
      </div>
    </div>
  );
}

export default App;
